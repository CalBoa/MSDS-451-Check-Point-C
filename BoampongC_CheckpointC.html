# 6. Performance metrics function
def performance_metrics(returns, benchmark=None):
    ann_return = returns.mean() * trading_days
    ann_vol = returns.std() * np.sqrt(trading_days)
    sharpe = ann_return / ann_vol if ann_vol != 0 else np.nan

    wealth = np.exp(returns.cumsum() / 100)
    running_max = wealth.cummax()
    drawdown = (wealth - running_max) / running_max
    max_dd = drawdown.min() * 100

    results = {
        "Annualized Return (%)": ann_return,
        "Annualized Volatility (%)": ann_vol,
        "Sharpe Ratio": sharpe,
        "Max Drawdown (%)": max_dd
    }

    if benchmark is not None:
        cov = np.cov(returns, benchmark)[0, 1]
        var_b = np.var(benchmark)
        beta = cov / var_b if var_b != 0 else np.nan
        alpha = ann_return - beta * (benchmark.mean() * trading_days)
        results["Alpha (%)"] = alpha
        results["Beta"] = beta

    return pd.Series(results)

# 8. Fee impact analysis
management_fees = [0.00, 0.01, 0.02, 0.04]  # annual fees
fee_results = []

for fee in management_fees:
    daily_fee = fee / trading_days * 100
    net_returns = portfolio_returns - daily_fee
    
    ann_return = net_returns.mean() * trading_days
    ann_vol = net_returns.std() * np.sqrt(trading_days)
    sharpe = ann_return / ann_vol if ann_vol != 0 else np.nan
    
    fee_results.append({
        "Management Fee (%)": fee * 100,
        "Net Annualized Return (%)": ann_return,
        "Net Sharpe Ratio": sharpe
    })

fee_table = pd.DataFrame(fee_results)

# 10. Monte Carlo simulation
mu = log_returns[portfolio_assets].mean().values
cov = log_returns[portfolio_assets].cov().values

n_days = 252 * 5
n_sims = 500
sim_end_values = []

for _ in range(n_sims):
    simulated = np.random.multivariate_normal(mu, cov, size=n_days)
    sim_portfolio_returns = simulated.dot(weights)
    sim_wealth = np.exp(sim_portfolio_returns.cumsum() / 100)
    sim_end_values.append(sim_wealth[-1])

sim_end_values = np.array(sim_end_values)

plt.figure(figsize=(10, 5))
plt.hist(sim_end_values, bins=30, edgecolor="black")
plt.title("Monte Carlo Distribution of 5-Year Ending Portfolio Value")
plt.xlabel("Ending Value")
plt.ylabel("Frequency")
plt.grid(True)
plt.show()
